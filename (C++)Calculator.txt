#include <iostream>
#include <string>
using namespace std;

int main()
{
	string arr;
	float a;
	float b;

	for (;;) {

		cout << "You can enter expression" << endl;
		cin >> a;
		cin >> arr;
		cin >> b;

		for (int i = 0; i <= arr.length(); i++)
			if (arr[i] == '*')
			{
				cout << a * b << endl;
			}
			else if (arr[i] == '+')
			{
				cout << a + b << endl;
			}
			else if (arr[i] == '-')
			{
				cout << a - b << endl;
			}
			else if (arr[i] == '/')
			{
				cout << a / b << endl;
			}	
	}

	cin.get();
}
